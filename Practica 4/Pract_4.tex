\section{Script de Octave que enumere todos los vectores}
\begin{verbatim}
n = 3;
count = 1;\
for i = 1:2^n\
  	binary = dec2bin(i-1,n);\
  	fprintf('Vector --d: [', count);\
  	for j = 1:n\
    	fprintf('%d', str2num(binary(j)));\
    	if j < n\
    		fprintf(', ');\
    	endif\
  	endfor\
  	fprintf('n');\
  	count = count + 1;\
endfor
\end{verbatim}
